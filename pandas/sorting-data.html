Import libraries

import numpy as np
import pandas as pd

Create DataFrame

data = {
    'name': ['Regina', 'Barbie', 'Gaz', 'Jiminy', 'Fran'],
    'number': np.random.randint(0, 100, 5)
}

df = pd.DataFrame(data)
df.head()
name number
0 Regina 40
1 Barbie 75
2 Gaz 84
3 Jiminy 68
4 Fran 23

Sorting our values

# Sorting in alphabetical order
df.sort_values('name')
name number
1 Barbie 75
4 Fran 23
2 Gaz 84
3 Jiminy 68
0 Regina 40 
# Sorting numbers in reverse order
df.sort_values('number', ascending=False)
name number
2 Gaz 84
1 Barbie 75
3 Jiminy 68
0 Regina 40
4 Fran 23

Sorting our index

df.set_index('name').sort_index(ascending=False)
number
name
